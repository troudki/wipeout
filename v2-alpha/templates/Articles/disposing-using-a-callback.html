Disposing using a callback

The method is used to register a callback which will be invoked when the viewModel itself is disposed of.

var myViewModel = wo.view.extend({ constructor: function myViewModel() { this._super(); }, onModelChanged: function(oldModel, newModel) { this._super(oldModel, newModel); if(newModel) { var eventRegistration = newModel.myObservable.subscribe(function() { // do something }); // ensure this event subscription is disposed of when this viewModel is disposed of this.registerDisposeCallback(function() { eventRegistration.dispose(); }); } } });
